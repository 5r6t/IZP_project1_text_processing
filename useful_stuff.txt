int main(int argc, char* argv[]) 
{
    (void) argc;
    (void) argv;
    char buffer[20];
    int c;
    for (int i = 0; c != '\n'; i++){
        c = fgetc(stdin);
        if (c == EOF){
            break;
        }
        buffer[i] = (char)c;
        buffer[i+1] = '\0';
    }
    printf("%s", buffer);
    return 0;
}
